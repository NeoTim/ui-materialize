HTML Setup

In Order for ui-materialize to work properly, we must wrap our application in the mz-materialize directive

If using side navs, wrap the application inside the mz-body directive, leaving the mz-nav-{side} on the outside

If using the nav-nav-bar, this remains inside the mz-body


        <html ng-app="app">
            <body>

                <mz-materialize>

                    <mz-body>
                        <mz-nav-bar view="nav-bar" fixed=true></mz-nav-bar>

                        <mz-main view></mz-main>

                        <mz-footer view="footer"></mz-footer>
                    </mz-body>

                    <mz-nav-right view="nav-right" viz="thin fat" fixed=true></mz-nav-right>

                </mz-materialize>


            </body>
        </html>

Here is how you can set up the ui-router to work with your views


       $stateProvider
           .state('app', {
               url: '/app',
               views: {
                   'nav-bar': { templateUrl: '/pathToNavBar.html'}
                   'nav-right': { templateUrl: '/pathToNavRight.html'}
                   'footer': { templateUrl: '/pathToFooter.html'}
               }
           })
           .state('app.home', {
               url: '/home',
               views: {
                   '@': { templateUrl: '/pathToHomeView.html'}
               }
           });
